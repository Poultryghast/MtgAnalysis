mdata <- download()

test <- function()
{
	source("cluster.txt")
	clustering <- allclusters(mdata[2:2])
	plot(table(clustering[[1]]))
}

cplot <- function()
{
	i <- 1
	while(TRUE)
	{
		plot(1:length(clustering[[i]]), lengths(clustering[[i]]), pch=16, main=mdata[[i]]$name)
		i <- i + 1
		if(i > length(clustering)){i = 1}
		readline(prompt="")
	}
}
